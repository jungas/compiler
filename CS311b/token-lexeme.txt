<program_declaration> -> program
<begin> -> begin
<end> -> end
<comma> -> ,
<integer> -> integer
<float> -> float
<boolean> -> boolean
<character> -> character
<character_string> -> string
<input_keyword> -> input
<print> -> print
<character_constant> -> '([^"]|\\")?'
<string_literal> -> "([^"]|\\")*"
<float_constant> -> [0-9]+\.[0-9]+
<integer_constant> -> [0-9]+
<boolean_constant> -> true|false
<logical_operator> -> and|or
<additive_operator> -> \+|\-
<multiplicative_operator> -> \*|/|mod
<relational_operator> -> <|>|==|!=|<=|=>
<assignment_operator> -> =
<if> -> if
<while> -> while
<program_name> -> [A-Z][a-zA-Z0-9_]*
<identifier> -> [a-zA-Z][a-zA-Z0-9_]*
<terminator> -> ;
<close_curly_brace> -> }
<open_curly_brace> -> \{
<open_parenthesis> -> \(
<close_parenthesis> -> \)
